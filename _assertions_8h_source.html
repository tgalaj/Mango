64#define MG_UNUSED(x) do { (void)sizeof(x); } while(0)

66#ifdef MG_ASSERTS_ENABLED
67#define MG_ASSERT(cond) \
68 do \
69 { \
70 if (!(cond)) \
71 { \
72 if (mango::asserts::reportFailure(#cond, __FILE__, __LINE__, 0) == \
73 mango::asserts::Halt) \
74 MG_HALT(); \
75 } \
76 } while(0)
77
78#define MG_ASSERT_MSG(cond, msg, ...) \
79 do \
80 { \
81 if (!(cond)) \
82 { \
83 if (mango::asserts::reportFailure(#cond, __FILE__, __LINE__, (msg), ##__VA_ARGS__) == \
84 mango::asserts::Halt) \
85 MG_HALT(); \
86 } \
87 } while(0)
88
89#define MG_ASSERT_FAIL(msg, ...) \
90 do \
91 { \
92 if (mango::asserts::reportFailure(0, __FILE__, __LINE__, (msg), ##__VA_ARGS__) == \
93 mango::asserts::Halt) \
94 MG_HALT(); \
95 } while(0)
96
97#define MG_VERIFY(cond) MG_ASSERT(cond)
98#define MG_VERIFY_MSG(cond, msg, ...) MG_ASSERT_MSG(cond, msg, ##__VA_ARGS__)
99#else
100#define MG_ASSERT(condition) \
101 do { MG_UNUSED(condition); } while(0)
102#define MG_ASSERT_MSG(condition, msg, ...) \
103 do { MG_UNUSED(condition); MG_UNUSED(msg); } while(0)
104#define MG_ASSERT_FAIL(msg, ...) \
105 do { MG_UNUSED(msg); } while(0)
106#define MG_VERIFY(cond) (void)(cond)
107#define MG_VERIFY_MSG(cond, msg, ...) \
108 do { (void)(cond); MG_UNUSED(msg); } while(0)
109#endif
